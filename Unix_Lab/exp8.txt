Program 1: Menu-Driven Calculator using case statement

case statement syntax:

case word in
pattern1)
    statements
    ;;
pattern2)
    statements
    ;;
*)
    default statements
    ;;
esac

Calculator Script:

#! /bin/sh
echo "Menu Driven Calculator"
echo "1. Addition"
echo "2. Subtraction"
echo "3. Multiplication"
echo "4. Division"
echo "Enter your choice: "
read ch
echo "Enter two numbers: "
read a
read b

case $ch in
1)
    sum=`expr $a + $b`
    echo "Sum = $sum"
    ;;
2)
    diff=`expr $a - $b`
    echo "Difference = $diff"
    ;;
3)
    mul=`expr $a \* $b`
    echo "Product = $mul"
    ;;
4)
    div=`expr $a / $b`
    echo "Division = $div"
    ;;
*)
    echo "Invalid Choice"
    ;;
esac

Program 2: Print Pattern

#! /bin/sh
echo "Enter number of rows: "
read rows

i=1
while [ $i -le $rows ]
do
    j=1
    while [ $j -le $i ]
    do
        echo -n "* "
        j=`expr $j + 1`
    done
    echo ""
    i=`expr $i + 1`
done

* * *

Program 3: Check if Directory/File Exists

File tests in Bash:

-f -> File exists
-d -> Directory exists
-e -> File or directory exists
!  -> NOT operator

Script to check and create directory:

#! /bin/sh
echo "Enter directory name: "
read dir

if [ -d "$dir" ]
then
    echo "Directory $dir already exists"
else
    mkdir $dir
    echo "Directory $dir created successfully"
fi

Script to check and create file:

#! /bin/sh
echo "Enter file name: "
read file

if [ -f "$file" ]
then
    echo "File $file already exists"
else
    touch $file
    echo "File $file created successfully"
fi
